# Generate WTS file 
python3 gen_wts.py -w own.pt -o own.wts
python3 gen_wts.py -w yolov5s.pt -o yolov5s.wts

# Cmake & Make 
# If using custom model, make sure to update kNumClas in yolov5/src/config.h
cd yolov5/
mkdir build
cd build

cp ../../own.wts .
cp ../../yolov5s.wts .

cmake ..
make 

# Build engine
./yolov5_det -s own.wts own.engine s
./yolov5_det -s yolov5s.wts yolov5s.engine s

# Test
./yolov5_det -d own.engine ../images
./yolov5_det -d yolov5s.engine ../images
